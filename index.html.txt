    <script>
        // CONFIGURACI√ìN SUPABASE - USA TUS CREDENCIALES REALES
        const supabaseUrl = 'https://svbhtidildglabhgqxwu.supabase.co';
        const supabaseKey = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InN2Ymh0aWRpbGRnbGFiaGdxeHd1Iiwicm9sZSI6ImFub24iLCJpYXQiOjE3NjE3NTM1MDksImV4cCI6MjA3NzMyOTUwOX0._HL69nSYzf81DNssZOD_3wmHU0MEyYiowX9FXdmfCfc';
        
        // Crear cliente Supabase
        const supabase = window.supabase.createClient(supabaseUrl, supabaseKey);

        // Cargar productos al iniciar
        document.addEventListener('DOMContentLoaded', function() {
            cargarProductos();
            
            // Buscar al presionar Enter
            document.getElementById('buscar').addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    buscarProductos();
                }
            });
        });

        async function cargarProductos() {
            try {
                const { data: productos, error } = await supabase
                    .from('productos')
                    .select('*')
                    .order('sku_principal');

                if (error) throw error;

                if (productos.length === 0) {
                    mostrarMensaje('No hay productos en la base de datos. Agrega algunos productos primero.');
                    return;
                }

                mostrarProductos(productos);
            } catch (error) {
                console.error('Error cargando productos:', error);
                mostrarError('Error al cargar productos: ' + error.message);
            }
        }

        async function buscarProductos() {
            const termino = document.getElementById('buscar').value.trim();
            
            try {
                let query = supabase
                    .from('productos')
                    .select('*');

                if (termino) {
                    query = query.or(
                        `sku_principal.ilike.%${termino}%,sku_secundario.ilike.%${termino}%,descripcion.ilike.%${termino}%,grupo.ilike.%${termino}%,familia.ilike.%${termino}%`
                    );
                } else {
                    query = query.order('sku_principal');
                }

                const { data: productos, error } = await query;

                if (error) throw error;

                if (productos.length === 0) {
                    mostrarMensaje('No se encontraron productos con ese criterio de b√∫squeda.');
                    return;
                }

                mostrarProductos(productos);
            } catch (error) {
                console.error('Error buscando productos:', error);
                mostrarError('Error en la b√∫squeda: ' + error.message);
            }
        }

        function mostrarProductos(productos) {
            const contenedor = document.getElementById('productos');
            
            if (productos.length === 0) {
                contenedor.innerHTML = '<div class="loading">No hay productos para mostrar</div>';
                return;
            }

            contenedor.innerHTML = productos.map(producto => `
                <div class="producto">
                    ${producto.imagen_url ? 
                        `<img src="${producto.imagen_url}" alt="${producto.descripcion}" onerror="this.style.display='none'">` : 
                        '<div style="text-align: center; color: #bdc3c7; margin: 20px 0;">üì∑ Sin imagen</div>'
                    }
                    <div class="sku-principal">${producto.sku_principal}</div>
                    <div class="sku-secundario">${producto.sku_secundario || 'Sin SKU secundario'}</div>
                    <div class="descripcion">${producto.descripcion}</div>
                    <div class="categorias">
                        ${producto.grupo ? `<span class="categoria">üì¶ ${producto.grupo}</span>` : ''}
                        ${producto.familia ? `<span class="categoria">üè∑Ô∏è ${producto.familia}</span>` : ''}
                    </div>
                    ${producto.marca ? `<div class="marca">üè≠ ${producto.marca}</div>` : ''}
                </div>
            `).join('');
        }

        function mostrarMensaje(mensaje) {
            const contenedor = document.getElementById('productos');
            contenedor.innerHTML = `<div class="loading">${mensaje}</div>`;
        }

        function mostrarError(mensaje) {
            const contenedor = document.getElementById('productos');
            contenedor.innerHTML = `<div class="error">${mensaje}</div>`;
        }
